USAGE INSTRUCTIONS FOR PHASE 3
================================================================================

1. Complete End-to-End Workflow:

   from parameter_extraction import ParameterExtractor
   from sis_network_propagation import SISNetworkPropagation
   from biomarker_validation import BiomarkerValidator

   # Extract parameters from disease modules
   extractor = ParameterExtractor(disease_modules, expr_data, clinical_data)
   params = extractor.extract_all_parameters(module_name, disease_group='disease_status')

   # Run SIS dynamics
   propagation = SISNetworkPropagation(
       adjacency_matrix=params['adjacency'],
       gene_names=module_genes,
       parameters={
           'beta': params['beta'],
           'gamma': params['gamma'],
           'initial_infection': params['initial_infection']
       }
   )
   propagation.run_dynamics(n_steps=500, n_runs=100)

   # Get biomarkers
   biomarkers_df = propagation.get_biomarker_table(top_n=20)
   
   # Validate predictions
   validator = BiomarkerValidator(
       predicted_biomarkers=biomarkers_df['gene'].tolist(),
       biomarker_scores=biomarkers_df['persistence'].values,
       expression_data=expr_data,
       clinical_data=clinical_data
   )
   expr_val = validator.validate_expression_changes('disease_status')
   clin_val = validator.validate_clinical_correlation('outcome')
